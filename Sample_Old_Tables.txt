CREATE TABLE `F_MICROOUTAGE_H` (
    `ENODEB_SECTOR_CARRIER` VARCHAR(50) COMMENT 'Concatenated Unique ID',
    `STARTDATE` DATE COMMENT 'Day of the outage',
    `HR` INT(2) COMMENT 'Hour of the outage',
    `TECHNOLOGY` VARCHAR(20) COMMENT 'Network technology (e.g., 5G, 4G)',
    `REGION` VARCHAR(50) COMMENT 'Geographic region',
    `MARKET_ID` VARCHAR(3) COMMENT 'Contains Market Code',
    `MARKET_DESC` VARCHAR(100) COMMENT 'Full description of the market',
	`SITE_NAME` VARCHAR(255) COMMENT 'site of the market',
    `GNODEB` VARCHAR(50) COMMENT 'gNodeB ID or name',
    `DU` INT(11) COMMENT 'Distributed Unit identifier',
    `SECTOR` INT(11) COMMENT 'Sector identifier',
    `CARRIER` INT(11) COMMENT 'Carrier/Frequency identifier',
    `CELLDOWNTIME` DECIMAL(18,4) COMMENT 'pmCellDowntimeAuto',
    `CELLDOWNLOCK` DECIMAL(18,4) COMMENT 'pmCellDownLockAuto',
    `UNPLANNED_DOWNTIME` DECIMAL(18,4) COMMENT 'unplanned_downtime',
    `DROP_%` DECIMAL(10,4) COMMENT 'Drop percentage',
    `DROP_%_NUM` DECIMAL(18,4) COMMENT 'Drop percentage Numerator',
    `DROP_%_DEN` DECIMAL(18,4) COMMENT 'Drop percentage Denominator',
    `4G_SAMSUNG_OUTAGE_STATE` VARCHAR(50) COMMENT 'Samsung 4G Outage State',
    `5G_S5NC_CRR_ALL` VARCHAR(50) COMMENT 'S5NC_CRR_All',
    `5G_CRR_NORMALRELEASE` VARCHAR(50) COMMENT 'S5NC_CRR_NormalRelease',
    `5GNR_CONTEXTDROP%_CRR_ABNORMAL_4G` VARCHAR(50) COMMENT '5GNR_ContextDrop%_CRR_abnormal_4G',
    `5G_SAMSUNG_OUTAGE_STATE` VARCHAR(50) COMMENT 'Samsung_Outage_State',    
    UNIQUE KEY `UNQ_OUTAGE` (`ENODEB_SECTOR_CARRIER`, `STARTDATE`, `HR`) USING HASH, 
    SHARD KEY `SHRD_OUTAGE` (`ENODEB_SECTOR_CARRIER`,`STARTDATE`),
    SORT KEY `SRT_OUTAGE` (`STARTDATE`, `MARKET_ID`),
    KEY `idx_filter` (`STARTDATE`, `MARKET_ID`) USING HASH
) COMMENT='Table tracking granular hourly network micro-outages';
